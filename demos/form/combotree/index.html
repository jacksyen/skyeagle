示例一（默认功能）

$('#combotree1').combotree({
	width : 200,		// combo的宽度设置
	url : 'tree_data.json',	// 结构树的json数据地址
	animate : true,		// 是够显示动画效果
	checkbox : false,	// 是否有多选框，次属性关系到是否能够多选
	cascadeCheck : false	// 是否级联选择
});

示例三（方法示例）

$('#combotree3').combotree({
	width : 200,		// combo的宽度设置
	url : 'tree_data.json',	// 结构树的json数据地址
	animate : true,		// 是够显示动画效果
	checkbox : true,	// 是否有多选框，次属性关系到是否能够多选
	cascadeCheck : true	// 是否级联选择
});
$('#destroy').click(function (){
	$('#combotree3').combotree('destroy');
});
$('#resize').click(function (){
	$('#combotree3').combotree('resize',{
		width : 300
	});
});
$('#disable').click(function (){
	$('#combotree3').combotree('disable');
});
$('#enable').click(function (){
	$('#combotree3').combotree('enable');
});
$('#getValue').click(function (){
	var val = $('#combotree3').combotree('getValue');
	alert('getVlaue :' + val);
});
$('#setValue').click(function (){
	var val = $('#combotree3').combotree('setValue','File1');
});
