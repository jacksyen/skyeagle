Examples

示例一（默认功能）

$('#autocomplete1').autocomplete({
	data : s		// 传入的数据
});

示例二（方法示例）

$('#autocomplete2').autocomplete({
	data : s		// 传入的数据
});
$('#destroy').click(function (){
	$('#autocomplete2').autocomplete('destroy');
});
$('#resize').click(function (){
	$('#autocomplete2').autocomplete('resize', 300);
});
$('#disable').click(function (){
	$('#autocomplete2').autocomplete('disable');
});
$('#enable').click(function (){
	$('#autocomplete2').autocomplete('enable');
});
