Examples

  • 默认功能

示例一（默认功能）

$('#validatebox1').validatebox({
	required : true,	// 验证能否为空
	tipwidth : null,	// 提示框宽度
	missMsg : '输入框不能为空'	// 验证失败提示信息
});

示例二（方法示例）

$('#validatebox2').validatebox({
	required : true,	// 验证能否为空
	tipwidth : null,	// 提示框宽度
	missMsg : '输入框不能为空'	// 验证失败提示信息
});
//判断是否验证成功
$('#validate').click(function (){
	var validate = $('#validatebox2').validatebox('validate');
	alert('validate:' + validate);
});
//清除数据
$('#clear').click(function (){
	$('#validatebox2').validatebox('clear');
});
//设置数据
$('#setValue').click(function (){
	var val = $('#validatebox2').validatebox('setValue','sdsadfasd');
});
