  • 默认功能

示例一（默认功能）

$('#tt').datagrid({
	// 显示分页栏
	pagination : true,
	// 显示每页记录数
	pageSize : 10,
	// 显示每页记录数列表
	pageList : [5,10],
	// width : 700,
	// height: 500,
	// 工具栏
	toolbar : [
		{type : 'button', options : {
			label : '选择全部', 
			onClick : function (){
				$('#tt').datagrid('selectAll');
			}}, line : true },
		{type : 'button', options : {
			label : '取消选择', 
			onClick : function (){
				$('#tt').datagrid('unselectAll');
			}}, line : true },
		{type : 'button', options : {
			label : '开始编辑第一条数据', 
			onClick : function (){
				$('#tt').datagrid('beginEdit', 0);
			}}, line : true },
		{type : 'button', options : {
			label : '结束编辑第一条数据', 
			onClick : function (){
				$('#tt').datagrid('endEdit', 0);
			}}, line : true },
		{type : 'button', options : {
			label : '获取选择项', 
			onClick : function (){
				var rows = $('#tt').datagrid('getSelections');
  				alert('rows.length:'+rows.length);
			}}, line : true },
		{type : 'button', options : {
			label : '追加行数据', 
			onClick : function (){
				$('#tt').datagrid('appendRow',{
					code : '999',
			  		name : 'new',
			  		addr : 'new addr'
				});
			}}, line : true },	
		{type : 'button', options : {
			label : '第三行插入行数据', 
			onClick : function (){
				$('#tt').datagrid('insertRow',{
					index : 2,
					row   : {
						code : '999',
				  		name : 'insertRow',
				  		addr : 'insertRow'
					}
				});
			}}, line : true },
		{type : 'button', options : {
			label : '隐藏Name列', 
			onClick : function (){
				$('#tt').datagrid('hideColumn','name');
			}}, line : true },
		{type : 'button', options : {
			label : '显示Name列', 
			onClick : function (){
				$('#tt').datagrid('showColumn','name');
			}}, line : true },
		{type : 'button', options : {
			label : '删除第一行', 
			onClick : function (){
				$('#tt').datagrid('deleteRow', 0);
			}}, line : true },
		{type : 'button', options : {
			label : '验证第一行', 
			onClick : function (){
				var result = $('#tt').datagrid('validateRow', 0);
				alert('validate result:' + result);
			}}, line : true },
		{type : 'button', options : {label : 'button2'} }
	],
	// 数据列配置对象
	columns:[[{
			title : '合并列',
			colspan : 2
		},{
			field:'code',
			title:'Code',
			width:80,
			rowspan: 2,
			editor : {type: 'numberbox' }
		}
	],[
		{
			field:'name',
			title:'Name',
			width:120,
			editor : {type: 'combo' }
		},
		{
			field:'addr',
			title:'Addr',
			width:80,
			align:'right',
			sortable : true,
			sorter : function(a, b) {
				return (a > b ? 1 : -1);
			}
		}
	]],
	// 行背景交替显示
	striped : true,
	// 标题
	title : 'datagrid title',
	// 显示行号
	rownumbers: true,
	// 初始化以'code'列排序
	sortName : 'code',
	// 排序为倒序
	sortOrder : 'desc',
	// 远程数据请求URL
	url : 'datagrid_data.json',
	onLoadSuccess : function (event, data) {
	  	
	},
	onBeforeLoad : function (event, param){
		return true;
	},
	onSortColumn : function (event, ui){
		alert('onSortColumn,sort:' + ui.sort +',order:' + ui.order);
	},
	onResizeColumn : function (event, ui){
		alert('onResizeColumn,field:' + ui.field +',width:' + ui.width);
	},
	onSelectAll : function (event, ui){
		alert('onSelectAll,ui.rows.length:' + ui.rows.length);
	},
	onUnselectAll : function (event, ui){
		alert('onUnselectAll,ui.rows.length:' + ui.rows.length);
	}
});
