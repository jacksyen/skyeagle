  • 默认功能

示例一（默认功能）

    $('#tt').tree({
	// 节点前面显示复选框
	checkbox : true,
	// 打开或关闭节点动画
	animate : true,
	// 支持拖放(此功能暂未实现)
	dnd : true,
	// 远程数据请求URL
	url : 'tree_data.json',
	// 双击节点事件
	onDblClick : function(event, node) {
		alert('onDblClick:' + 'id:' + node.id + ',text:' + node.text + ',checked:' + node.checked + ',attributes:' + node.attributes + ',target:' + node.target);
	},
	// 右击节点事件
	onContextMenu : function(event, node) {
		alert('onContextMenu:' + 'id:' + node.id + ',text:' + node.text + ',checked:' + node.checked + ',attributes:' + node.attributes + ',target:' + node.target);
	}
});
$('#beginEdit').button({
	onClick : function (){
		// 编辑节点
		var node = $('#tt').tree('getSelected');
		if(!node){
			alert('没有选择项');
			return;
		}
		$('#tt').tree('beginEdit', node.target);
	}
});
